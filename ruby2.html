👤 User

class User
  @@user_count = 0
  def initialize name = ""
      # construtor        variáveis
      @username = name  # ├─ instância
      @@user_count += 1 # └─ classe
  end

#[...]
  def self.count
    @@user_count
  end

#[...]
  def get_name 
    @username
  end
  
  def set_name! name
    @username = name
  end

#[...] Getter & Setter
  def username # <User>.username
    @username
  end
  # <User>.username = "novo nome"
  def username=(name)
    @username = name
  end

attr_writer :var, attr_reader :var e attr_accessor :var
